\section{Responsabilité du commandement}

\begin{e1}
	\itemt{\gls{gc} soutenu par le \gls{cas}}{
		Les intentions du \gls{gc} doivent être clairement définies.
		
		Le \gls{gc} doit s'assurer que les planificateurs comprennent l'objectif, les besoins du \gls{cc}, et les \glspl{roe} spécifiques.
		
		Le \gls{gc} fournit également un cadre de travail pour l'évaluation des risques lors du \gls{tac}.
	}
	
	\itemt{Renseignement}{
		Le renseignement est responsable du \gls{jipoe} et fournit les dernières informations quant aux cibles, les images obtenues par reconnaissance, \ldots{}.
	}
	
	\itemt{Opérations}{
		Le S3 est responsable de s'assurer que le \gls{cas} est correctement intégré à l'\gls{opord} et au plan d'appui-feu.
	}
	
	\itemt{\gls{alo}}{
		L'\gls{alo} conseille le \gls{gc} pour l'utilisation du \gls{cas}.
	}
	
\end{e1}
